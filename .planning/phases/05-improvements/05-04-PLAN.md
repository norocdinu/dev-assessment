1. Add page state and total state after the existing `columnFilters` state declaration:
```tsx
  const [page, setPage] = useState(1);
  const [total, setTotal] = useState(0);
  const PAGE_SIZE = 25;
```

2. Replace the existing `fetchSubmissions` function to include page params and handle the new response shape:
```tsx
  async function fetchSubmissions(currentPage = page) {
    setLoading(true);
    try {
      const params = new URLSearchParams();
      if (filterTestConfigId) params.set('testConfigId', filterTestConfigId);
      if (filterDateFrom) params.set('dateFrom', filterDateFrom);
      if (filterDateTo) params.set('dateTo', filterDateTo);
      if (filterDifficulty) params.set('difficulty', filterDifficulty);
      params.set('page', String(currentPage));
      params.set('pageSize', String(PAGE_SIZE));
      const res = await api.get(`/admin/submissions?${params}`);
      setSubmissions(res.data.data);
      setTotal(res.data.total);
    } finally {
      setLoading(false);
    }
  }
```

3. Update `applyFilters` to reset page to 1 before fetching:
```tsx
  function applyFilters() {
    setPage(1);
    fetchSubmissions(1);
    setSelectedIds(new Set());
  }
```

4. Update `clearFilters` to reset page to 1:
```tsx
  function clearFilters() {
    setFilterTestConfigId('');
    setFilterDateFrom('');
    setFilterDateTo('');
    setFilterDifficulty('');
    setSelectedIds(new Set());
    setPage(1);
    setTimeout(() => fetchSubmissions(1), 0);
  }
```

5. Add a `handlePageChange` function after `clearFilters`:
```tsx
  function handlePageChange(newPage: number) {
    setPage(newPage);
    fetchSubmissions(newPage);
  }
```

6. Update the initial `useEffect` to call `fetchSubmissions(1)` explicitly (since the function signature now has a param with default):
```tsx
  useEffect(() => {
    api.get('/test-configs').then((r) => setTestConfigs(r.data)).catch(() => {});
    fetchSubmissions(1);
  }, []);
```

Note: Because `fetchSubmissions` uses `filterTestConfigId` etc. from closure and they start as empty strings on mount, this is safe. The `eslint-disable` comment for exhaustive-deps may be needed if linting flags it — add `// eslint-disable-next-line react-hooks/exhaustive-deps` above the `useEffect` if required.
