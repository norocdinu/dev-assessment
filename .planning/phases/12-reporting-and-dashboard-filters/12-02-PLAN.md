**Step 4 — add computeDateRange helper function** immediately after the state declarations (before any useEffect):
```typescript
function computeDateRange(): { from?: string; to?: string } {
  const now = new Date();
  if (datePreset === '7d') {
    const d = new Date(now); d.setDate(d.getDate() - 7);
    return { from: d.toISOString(), to: now.toISOString() };
  }
  if (datePreset === '30d') {
    const d = new Date(now); d.setDate(d.getDate() - 30);
    return { from: d.toISOString(), to: now.toISOString() };
  }
  if (datePreset === '90d') {
    const d = new Date(now); d.setDate(d.getDate() - 90);
    return { from: d.toISOString(), to: now.toISOString() };
  }
  if (datePreset === 'custom') {
    return { from: customFrom || undefined, to: customTo || undefined };
  }
  return {};
}
```

With a `useCallback`-wrapped `fetchData` plus a reactive `useEffect`:
```typescript
const fetchData = useCallback(() => {
  // Guard: if custom preset but dates not yet both filled, skip fetch
  if (datePreset === 'custom' && (!customFrom || !customTo)) return;

  setLoading(true);
  setError('');
  const dateRange = computeDateRange();
  const params = {
    ...(testConfigId ? { testConfigId } : {}),
    ...(dateRange.from ? { from: dateRange.from } : {}),
    ...(dateRange.to   ? { to: dateRange.to }     : {}),
  };

  Promise.all([
    api.get('/dashboard/stats',      { params }),
    api.get('/dashboard/competency', { params }),
  ])
    .then(([statsRes, competencyRes]) => {
      setStats(statsRes.data);
      setCompetency(competencyRes.data);
    })
    .catch(() => setError('Failed to load dashboard data. Try again.'))
    .finally(() => setLoading(false));
}, [testConfigId, datePreset, customFrom, customTo]);

useEffect(() => {
  fetchData();
}, [fetchData]);
```

Note: `computeDateRange` is called inside `fetchData` so it captures the current state values via closure at call time. `fetchData` is itself in the useCallback dependency array so it re-creates whenever filter state changes, which in turn triggers the useEffect.
