# Plan 04-02: CSV Export API + Candidate Comparison API

## Objective
Add `GET /admin/submissions/export` for CSV file download and `GET /admin/submissions/compare` for fetching multiple submission details in parallel. Both routes must be registered before the existing `/:linkId` wildcard route. Without correct ordering Fastify will match the literal strings "export" and "compare" as linkId param values, returning 404 "Submission not found".

  // GET /admin/submissions/export
  app.get('/export', { preHandler: [authMiddleware] }, async (request, reply) => {
    const queryRaw = request.query as Record<string, string>;
    const testConfigId = queryRaw.testConfigId;

    const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
    if (!testConfigId || !uuidRegex.test(testConfigId)) {
      return reply.status(400).send({ error: 'testConfigId (UUID) is required' });
    }

    const rows = await db`
      SELECT
        sr.score_pct,
        sr.pass,
        sr.time_taken_seconds,
        tl.submitted_at,
        tc.name        AS test_name,
        tc.difficulty,
        t.name         AS technology_name
      FROM submission_results sr
      JOIN test_links   tl ON tl.id = sr.link_id
      JOIN test_configs tc ON tc.id = tl.test_config_id
      JOIN technologies t  ON t.id  = tc.technology_id
      WHERE tl.test_config_id = ${testConfigId}
      ORDER BY tl.submitted_at DESC
    `;

    const escape = (v: string | number | boolean) =>
      `"${String(v).replace(/"/g, '""')}"`;

    const headers = ['Test Name', 'Technology', 'Difficulty', 'Score %', 'Pass/Fail', 'Time (seconds)', 'Submitted At'];
    const dataRows = (rows as Array<{
      test_name: string; technology_name: string; difficulty: string;
      score_pct: number; pass: boolean; time_taken_seconds: number; submitted_at: string;
    }>).map(s => [
      escape(s.test_name),
      escape(s.technology_name),
      escape(s.difficulty),
      escape(s.score_pct),
      escape(s.pass ? 'Pass' : 'Fail'),
      escape(s.time_taken_seconds),
      escape(s.submitted_at),
    ].join(','));

    const csv = [headers.map(h => escape(h)).join(','), ...dataRows].join('\r\n');

    reply.header('Content-Type', 'text/csv');
    reply.header('Content-Disposition', `attachment; filename="submissions-${testConfigId}.csv"`);
    return reply.send(csv);
  });

This route must appear BEFORE GET /compare, GET /, and GET /:linkId in the function body.

  // GET /admin/submissions/compare?ids=linkId1,linkId2[,linkId3,linkId4]
  app.get('/compare', { preHandler: [authMiddleware] }, async (request, reply) => {
    const queryRaw = request.query as Record<string, string>;
    const idsParam = queryRaw.ids ?? '';
    const ids = idsParam.split(',').map((s: string) => s.trim()).filter(Boolean);

    const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
    if (ids.length < 2) {
      return reply.status(400).send({ error: 'At least 2 link IDs required' });
    }
    if (!ids.every((id: string) => uuidRegex.test(id))) {
      return reply.status(400).send({ error: 'All IDs must be valid UUIDs' });
    }

    const results = await Promise.all(ids.map(async (linkId: string) => {
      const [result] = await db`
        SELECT
          sr.score_pct,
          sr.pass,
          sr.skill_area_scores,
          sr.time_taken_seconds,
          sr.graded_at,
          tl.id          AS link_id,
          tl.submitted_at,
          tl.test_config_id,
          tc.pass_threshold_pct,
          tc.name AS test_name,
          t.name  AS technology_name,
          tc.difficulty
        FROM submission_results sr
        JOIN test_links tl      ON tl.id = sr.link_id
        JOIN test_configs tc    ON tc.id = tl.test_config_id
        JOIN technologies t     ON t.id  = tc.technology_id
        WHERE tl.id = ${linkId}
      `;
      return result ?? null;
    }));

    const missingIndex = results.findIndex(r => r === null);
    if (missingIndex !== -1) {
      return reply.status(404).send({ error: `Submission not found: ${ids[missingIndex]}` });
    }

    return reply.status(200).send(results);
  });

The route must appear BEFORE GET / and GET /:linkId in the function body.
