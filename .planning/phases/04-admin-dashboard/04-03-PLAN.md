# Plan 04-03: Bulk Question Import API

## Objective
Install `@fastify/multipart` (not currently in package.json), register it as a Fastify plugin, and implement `POST /questions/import` — an owner-only endpoint that accepts a CSV file upload, validates each row against the existing question schema, inserts valid rows into the questions table, and returns a structured result with per-row error detail for invalid rows. Invalid rows are skipped; the import is partial.

Edit apps/api/src/routes/questions.ts. Add the following route at the END of the questionRoutes function body (after the DELETE route), before the closing brace:

  // POST /questions/import - bulk CSV import (owner only)
  app.post('/import', { preHandler: [authMiddleware, requireRole('owner')] }, async (request, reply) => {
    const data = await request.file();
    if (!data) return reply.status(400).send({ error: 'No file uploaded' });

    const buffer = await data.toBuffer();
    const text = buffer.toString('utf-8');

    // Split lines, remove BOM if present, filter blanks
    const lines = text
      .replace(/^﻿/, '')
      .split(/\r?\n/)
      .map((l: string) => l.trim())
      .filter(Boolean);

    if (lines.length < 2) {
      return reply.status(400).send({ error: 'CSV must have a header row and at least one data row' });
    }

    const EXPECTED_COLS = 10;

    function parseCSVLine(line: string): string[] {
      const result: string[] = [];
      let current = '';
      let inQuotes = false;
      for (let i = 0; i < line.length; i++) {
        const ch = line[i];
        if (ch === '"') {
          if (inQuotes && line[i + 1] === '"') {
            current += '"';
            i++;
          } else {
            inQuotes = !inQuotes;
          }
        } else if (ch === ',' && !inQuotes) {
          result.push(current);
          current = '';
        } else {
          current += ch;
        }
      }
      result.push(current);
      return result;
    }

    const [, ...dataLines] = lines; // skip header row
    let imported = 0;
    const errors: { row: number; reason: string }[] = [];

    // Load all technologies once for slug to id lookup
    const technologies = await db`SELECT id, slug FROM technologies`;
    const techMap = new Map<string, string>(
      technologies.map((t: { id: string; slug: string }) => [t.slug, t.id])
    );

    for (let i = 0; i < dataLines.length; i++) {
      const rowNum = i + 2; // 1-based, row 1 is header
      const cols = parseCSVLine(dataLines[i]);

      if (cols.length < EXPECTED_COLS) {
        errors.push({ row: rowNum, reason: `Expected ${EXPECTED_COLS} columns, got ${cols.length}` });
        continue;
      }

      const [technology_slug, difficulty, skill_area, text,
             option_a, option_b, option_c, option_d,
             correct_option, explanation] = cols.map((c: string) => c.trim());

      const technology_id = techMap.get(technology_slug);
      if (!technology_id) {
        errors.push({ row: rowNum, reason: `Unknown technology: '${technology_slug}'` });
        continue;
      }

      const rowSchema = questionBodySchema.safeParse({
        technology_id,
        difficulty,
        skill_area,
        text,
        option_a,
        option_b,
        option_c,
        option_d,
        correct_option,
        explanation: explanation || undefined,
      });

      if (!rowSchema.success) {
        const firstError = rowSchema.error.errors[0];
        errors.push({ row: rowNum, reason: `${firstError.path.join('.')}: ${firstError.message}` });
        continue;
      }

      const familyId = uuidv4();

      await db`
        INSERT INTO questions (
          family_id, version, technology_id, difficulty, skill_area,
          text, option_a, option_b, option_c, option_d, correct_option,
          explanation, created_by
        ) VALUES (
          ${familyId}, 1, ${rowSchema.data.technology_id}, ${rowSchema.data.difficulty},
          ${rowSchema.data.skill_area}, ${rowSchema.data.text}, ${rowSchema.data.option_a},
          ${rowSchema.data.option_b}, ${rowSchema.data.option_c}, ${rowSchema.data.option_d},
          ${rowSchema.data.correct_option}, ${rowSchema.data.explanation ?? null}, ${request.user.id}
        )
      `;
      imported++;
    }

    return reply.status(200).send({ imported, errors });
  });

Note: uuidv4 is already imported at the top of questions.ts as `import { v4 as uuidv4 } from 'uuid'` - use that existing import directly. questionBodySchema is already defined at the top of the file - do not redefine it.
