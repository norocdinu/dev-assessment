  'use client';

  import { useState, useEffect } from 'react';
  import { useSearchParams, useRouter } from 'next/navigation';
  import { api } from '@/lib/api';

  interface CandidateResult {
    link_id: string;
    test_name: string;
    technology_name: string;
    difficulty: string;
    score_pct: number;
    pass: boolean;
    time_taken_seconds: number;
    submitted_at: string;
    skill_area_scores: Record<string, { correct: number; total: number; pct: number }>;
  }

  function formatTime(seconds: number): string {
    const m = Math.floor(seconds / 60);
    const s = seconds % 60;
    return `${m}m ${s.toString().padStart(2, '0')}s`;
  }

  export default function ComparePage() {
    const searchParams = useSearchParams();
    const router = useRouter();
    const idsParam = searchParams.get('ids') ?? '';
    const ids = idsParam.split(',').map(s => s.trim()).filter(Boolean);

    const [candidates, setCandidates] = useState<CandidateResult[]>([]);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState('');

    useEffect(() => {
      if (ids.length < 2) return;
      setLoading(true);
      setError('');
      api.get(`/admin/submissions/compare?ids=${ids.join(',')}`)
        .then(r => setCandidates(r.data))
        .catch(() => setError('Failed to load comparison data. Please try again.'))
        .finally(() => setLoading(false));
    }, [idsParam]);

    if (ids.length < 2) {
      return (
        <div className="p-6">
          <h2 className="text-xl font-semibold text-gray-900 mb-4">Compare Candidates</h2>
          <p className="text-sm text-gray-400">Select at least 2 submissions to compare.</p>
          <button
            onClick={() => router.push('/submissions')}
            className="mt-4 text-sm text-blue-600 hover:underline"
          >
            Back to Submissions
          </button>
        </div>
      );
    }

    if (loading) {
      return (
        <div className="p-6">
          <h2 className="text-xl font-semibold text-gray-900 mb-4">Compare Candidates</h2>
          <p className="text-sm text-gray-400">Loading...</p>
        </div>
      );
    }

    if (error) {
      return (
        <div className="p-6">
          <h2 className="text-xl font-semibold text-gray-900 mb-4">Compare Candidates</h2>
          <p className="text-sm text-red-600">{error}</p>
          <button
            onClick={() => router.push('/submissions')}
            className="mt-4 text-sm text-blue-600 hover:underline"
          >
            Back to Submissions
          </button>
        </div>
      );
    }

    // Collect all unique skill areas across all candidates, sorted alphabetically
    const allSkillAreas = Array.from(
      new Set(candidates.flatMap(c => Object.keys(c.skill_area_scores)))
    ).sort();

    return (
      <div className="p-6">
        <div className="flex items-center justify-between mb-6">
          <h2 className="text-xl font-semibold text-gray-900">Compare Candidates</h2>
          <button
            onClick={() => router.push('/submissions')}
            className="text-sm text-blue-600 hover:underline"
          >
            Back to Submissions
          </button>
        </div>

        <div className="overflow-x-auto">
          <div className="flex gap-6 min-w-max">
            {candidates.map((candidate, idx) => {
              const passClass = candidate.pass
                ? 'bg-green-100 text-green-700'
                : 'bg-red-100 text-red-700';
              return (
                <div
                  key={candidate.link_id}
                  className="w-72 bg-white border border-gray-200 rounded-md p-6 flex-shrink-0"
                >
                  <div className="mb-4">
                    <div className="text-xs text-gray-500 mb-1">Candidate {idx + 1}</div>
                    <div className="text-sm font-medium text-gray-900">{candidate.test_name}</div>
                    <div className="text-xs text-gray-500 capitalize">
                      {candidate.technology_name} — {candidate.difficulty}
                    </div>
                  </div>

                  <div className="space-y-3 mb-6">
                    <div className="flex items-center justify-between">
                      <span className="text-xs text-gray-500">Score</span>
                      <span className="text-lg font-semibold text-gray-900">{candidate.score_pct}%</span>
                    </div>
                    <div className="flex items-center justify-between">
                      <span className="text-xs text-gray-500">Result</span>
                      <span className={`inline-flex items-center px-2 py-0.5 rounded-full text-xs font-medium ${passClass}`}>
                        {candidate.pass ? 'Pass' : 'Fail'}
                      </span>
                    </div>
                    <div className="flex items-center justify-between">
                      <span className="text-xs text-gray-500">Time taken</span>
                      <span className="text-sm text-gray-700">{formatTime(candidate.time_taken_seconds)}</span>
                    </div>
                    <div className="flex items-center justify-between">
                      <span className="text-xs text-gray-500">Submitted</span>
                      <span className="text-xs text-gray-500">
                        {new Date(candidate.submitted_at).toLocaleDateString('en-GB', {
                          day: 'numeric', month: 'short', year: 'numeric',
                        })}
                      </span>
                    </div>
                  </div>

                  {allSkillAreas.length > 0 && (
                    <div>
                      <div className="text-xs font-medium text-gray-600 mb-2">Skill Breakdown</div>
                      <div className="space-y-1">
                        {allSkillAreas.map(area => {
                          const score = candidate.skill_area_scores[area];
                          return (
                            <div key={area} className="flex items-center justify-between text-xs">
                              <span className="text-gray-600 truncate max-w-[140px]" title={area}>
                                {area}
                              </span>
                              <span className="text-gray-700 ml-2">
                                {score ? `${score.correct}/${score.total} (${score.pct}%)` : '-'}
                              </span>
                            </div>
                          );
                        })}
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>
        </div>
      </div>
    );
  }

  async function handleImport(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (!file) return;
    e.target.value = '';  // reset so same file can be re-selected

    setImporting(true);
    setImportResult(null);
    try {
      const formData = new FormData();
      formData.append('file', file);
      const res = await api.post('/questions/import', formData, {
        headers: { 'Content-Type': 'multipart/form-data' },
      });
      setImportResult(res.data);
      fetchQuestions();
    } catch {
      setImportResult({ imported: 0, errors: [{ row: 0, reason: 'Upload failed. Please try again.' }] });
    } finally {
      setImporting(false);
    }
  }
